NP V NP S_ING (NP-P-ING-SC; on-PP)
  Syntax: NP[Agent] VERB NP[Asset] PREP[on] NP[Goal +sc_ing]
  Semantics:
    * exist(start(E), Asset)
    * exist(end(E), Asset)
    * spend(during(E), Agent, Asset, Goal)
NP V NP PP.goal (NP-PP; on-PP)
  Syntax: NP[Agent] VERB NP[Asset] PREP[on] NP[Goal -sentential]
  Semantics:
    * exist(start(E), Asset)
    * exist(end(E), Asset)
    * spend(during(E), Agent, Asset, Goal)
NP V NP PP.theme (NP-PP; for-PP)
  Syntax: NP[Agent] VERB NP[Asset] PREP[for] NP[Theme]
  Semantics:
    * transfer(during(E), Theme)
    * has_possession(start(E), Agent, Asset)
    * has_possession(end(E), Agent, Theme)
    * value(E, Theme, Asset)
NP V NP ((SUBCAT MP))
  Syntax: NP[Agent] VERB NP[Asset]
  Semantics:
    * transfer(during(E), ?Theme)
    * has_possession(start(E), Agent, Asset)
    * has_possession(end(E), Agent, ?Theme)
    * value(E, ?Theme, Asset)
