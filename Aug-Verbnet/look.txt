  Syntax: NP[Experiencer] VERB PREP[+spatial] NP[Stimulus]
  Semantics:
    * perceive(during(E), Experiencer, Stimulus)
    * in_reaction_to(E, Stimulus)
NP V PP.location (PP; Location-PP)
  Syntax: NP[Agent] VERB PREP[over through into] NP[Location]
  Semantics:
    * search(during(E), Agent, Location, ?Theme)
NP V PP.location PP.theme (PP-PP; Location-PP Theme-PP)
  Syntax: NP[Agent] VERB PREP[+loc] NP[Location] PREP[for] NP[Theme]
  Semantics:
    * search(during(E), Agent, Location, Theme)
NP V PP.attribute (PP-PRED-RSin-PP)
  Syntax: NP[Theme] VERB PREP[in] NP[Attribute -sentential]
  Semantics:
    * seem(E, Theme, Attribute)
