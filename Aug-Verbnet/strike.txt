NP V NP (Basic Transitive)
  Syntax: NP[Stimulus] VERB NP[Experiencer]
  Semantics:
    * cause(Stimulus, E)
    * emotional_state(result(E), Emotion, Experiencer)
NP V ADV-Middle (Middle Construction)
  Syntax: NP[Experiencer] VERB ADV
  Semantics:
    * emotional_state(result(E), ?Emotion, Experiencer)
NP V NP-PRO-ARB (PRO-Arb Object Alternation)
  Syntax: NP[Stimulus] VERB
  Semantics:
    * cause(Stimulus, E)
    * emotional_state(result(E), Emotion, ?Experiencer)
NP.cause V NP (NP; Attribute Subject)
  Syntax: NP[Stimulus +genitive] LEX['s] VERB NP[Experiencer]
  Semantics:
    * cause(Stimulus, E)
    * emotional_state(result(E), Emotion, Experiencer)
NP V NP ADJ (NP-ADJP; Resultative)
  Syntax: NP[Stimulus] VERB NP[Experiencer] NP[Result]
  Semantics:
    * cause(Stimulus, E)
    * emotional_state(result(E), Emotion, Experiencer)
    * Pred(result(E), Experiencer)
NP V together (Together Reciprocal Alternation Intransitive)
  Syntax: NP[Theme +plural] VERB LEX[together]
  Semantics:
    * manner(during(E), directedmotion, Theme_i)
    * manner(during(E), directedmotion, Theme_j)
    * manner(end(E), forceful, Theme_i)
    * manner(end(E), forceful, Theme_j)
    * contact(end(E), Theme_i, Theme_j)
NP V PP (Conative)
  Syntax: NP[Agent] VERB LEX[at] NP[Patient]
  Semantics:
    * cause(Agent, E)
    * manner(during(E), directedmotion, Agent)
    * contact(during(E), Agent, Patient)
NP V PP PP (Conative)
  Syntax: NP[Agent] VERB LEX[at] NP[Patient] PREP[with] NP[Instrument]
  Semantics:
    * cause(Agent, E)
    * manner(during(E), directedmotion, Instrument)
    * contact(during(E), Instrument, Patient)
    * use(during(E), Agent, Instrument)
NP V (Basic Intransitive)
  Syntax: NP[Theme] VERB
  Semantics:
    * emit(during(E), Theme, Sound)
NP V PP.location (PP; Location-PP)
  Syntax: NP[Theme] VERB PREP[+loc] NP[Location]
  Semantics:
    * emit(during(E), Theme, Sound)
    * Prep(during(E), Sound, Location)
NP.location V PP.theme (PP; Location Subject, with-PP)
  Syntax: NP[Location] VERB PREP[with] NP[Theme]
  Semantics:
    * emit(during(E), Theme, Sound)
    * location(during(E), Sound, Location)
PP.location V NP (Locative Inversion)
  Syntax: PREP[+loc] NP[Location] VERB NP[Theme]
  Semantics:
    * emit(during(E), Theme, Sound)
    * Prep(during(E), Sound, Location)
There V NP PP (There-insertion)
  Syntax: LEX[there] VERB NP[Theme] PREP[+loc] NP[Location]
  Semantics:
    * emit(during(E), Theme, Sound)
    * Prep(during(E), Sound, Location)
NP V NP.theme (NP; Causative variant)
  Syntax: NP[Agent] VERB NP[Theme]
  Semantics:
    * emit(during(E), Theme, Sound)
    * cause(Agent, E)
