NP V (Basic Intransitive)
  Syntax: NP[Theme] VERB
  Semantics:
    * exist(during(E), Theme)
NP V PP.topic (PP; on-PP)
  Syntax: NP[Agent] VERB PREP[on] NP[Topic -sentential]
  Semantics:
    * think(during(E), Agent, Topic)
    * cause(E, Agent)
NP V PP.theme S_ING (P-ING-SC; on-PP)

    * think(during(E), Agent, Topic)
    * cause(E, Agent)

NP V PP.topic (PP; on-PP)
  Syntax: NP[Agent] VERB PREP[about] NP[Topic -sentential]
  Semantics:
    * transfer_info(during(E), Agent, ?Recipient, Topic)
    * cause(Agent, E)

NP V PP.location (PP; Location-PP)
  Syntax: NP[Theme] VERB PREP[at in with] NP[Location]
  Semantics:
    * location(start(E), Theme, Location)
    * location(during(E), Theme, Location)
NP V ADV (ADVP-PRED; here/there)
  Syntax: NP[Theme] VERB NP[Location +adv_loc]
  Semantics:
    * location(start(E), Theme, Location)
    * location(during(E), Theme, Location)
