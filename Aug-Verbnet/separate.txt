NP V NP (NP)
  Syntax: NP[Agent] VERB NP[Theme]
  Semantics:
    * group(during(E), Theme)
    * cause(Agent, E)
NP V NP PP.goal (NP-PP; as-PP)
  Syntax: NP[Agent] VERB NP[Theme] PREP[as] NP[Goal -sentential]
  Semantics:
    * group(during(E), Theme, Goal)
    * cause(Agent, E)
NP V NP PP.location (NP-PP; in-PP)
  Syntax: NP[Agent] VERB NP[Theme] PREP[in] NP[Goal]
  Semantics:
    * group(during(E), Theme, Goal)
    * cause(Agent, E)
NP V NP (Basic Transitive)
  Syntax: NP[Agent] VERB NP[Theme]
  Semantics:
    * cause(Agent, E)
    * location(start(E), Theme, ?Source)
    * location(end(E), Theme, ?Source)
NP V NP PP.source (NP-PP; Source-PP)
  Syntax: NP[Agent] VERB NP[Theme] PREP[+src] NP[Source]
  Semantics:
    * cause(Agent, E)
    * location(start(E), Theme, Source)
    * location(end(E), Theme, Source)
NP V PP.co-patient (PP; with-PP)
  Syntax: NP[Patient] VERB PREP[with] NP[Co-Patient]
  Semantics:
    * together(start(E), physical/abstract, Patient, Co-Patient)
    * apart(end(E), physical/abstract, Patient, Co-Patient)
