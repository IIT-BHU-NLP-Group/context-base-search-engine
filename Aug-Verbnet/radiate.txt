NP V (Basic Intransitive)
  Syntax: NP[Source] VERB
  Semantics:
    * emit(during(E), Source, ?Theme)
NP V NP (Transitive)
  Syntax: NP[Source] VERB NP[Theme]
  Semantics:
    * emit(during(E), Source, Theme)
NP.theme V PP.source (PP; Substance Subject, Source-PP)
  Syntax: NP[Theme] VERB PREP[from out_of] NP[Source]
  Semantics:
    * emit(during(E), Source, Theme)
NP V PP.location (NP-PP; Location-PP)
  Syntax: NP[Theme] VERB PREP[+path] NP[Location]
  Semantics:
    * emit(during(E), ?Source, Theme)
    * Prep(during(E), Theme, Location)
NP.location V PP.theme (PP; Location Subject, with-PP)
  Syntax: NP[Location] VERB PREP[with] NP[Theme]
  Semantics:
    * emit(during(E), ?Source, Theme)
    * location(during(E), Theme, Location)
PP.location V NP (Locative Inversion)
  Syntax: PREP[+loc] NP[Location] VERB NP[Theme]
  Semantics:
    * emit(during(E), ?Source, Theme)
    * Prep(during(E), Theme, Location)
There V NP PP (NP-PP; Expletive-there Subject)
  Syntax: LEX[there] VERB NP[Theme] PREP[+loc] NP[Location]
  Semantics:
    * emit(during(E), ?Source, Theme)
    * Prep(during(E), Theme, Location)
NP V out (PARTout)
  Syntax: NP[Theme] VERB LEX[out]
  Semantics:
    * emit(during(E), ?Source, Theme)

  Syntax: NP[Source] VERB NP[Theme]
