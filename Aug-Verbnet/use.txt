NP V NP (Basic Transitive)
  Syntax: NP[Agent] VERB NP[Asset]
  Semantics:
    * exist(start(E), Asset)
    * exist(end(E), Asset)
    * spend(during(E), Agent, Asset, ?Goal)
NP V NP S_ING (NP-ING-SC)
  Syntax: NP[Agent] VERB NP[Asset] NP[Goal +sc_ing]
  Semantics:
    * exist(start(E), Asset)
    * exist(end(E), Asset)
    * spend(during(E), Agent, Asset, Goal)
NP V (Basic Intransitive)
  Syntax: NP[Agent] VERB
  Semantics:
    * take_in(during(E), Agent, ?Patient)
NP V NP (Basic Transitive)
  Syntax: NP[Agent] VERB NP[Patient]
  Semantics:
    * take_in(during(E), Agent, ?Patient)
NP.location V NP (Location Subject Alternation)
  Syntax: NP[Location] VERB NP[Value]
  Semantics:
    * capacity(E, Location, Value)
NP V NP PP.location (NP-PP; Location-PP)
  Syntax: NP[Agent] VERB NP[Value] PREP[in] NP[Location]
  Semantics:
    * capacity(E, Location, Value)
    * cause(Agent, E)
NP V NP (Basic Transitive)
  Syntax: NP[Agent] VERB NP[Theme]
  Semantics:
    * use(during(E), Agent, Theme)
NP V NP ADV (NP-ADVP)
  Syntax: NP[Agent] VERB NP[Theme] ADV
  Semantics:
    * use(during(E), Agent, Theme, Predicate)
NP V NP PP.predicate (NP-PP; for-PP)
  Syntax: NP[Agent] VERB NP[Theme] PREP[for] NP[Predicate]
  Semantics:
    * use(during(E), Agent, Theme, Predicate)
NP V NP S_INF (NP-TO-INF-VC)
  Syntax: NP[Agent] VERB NP[Theme] NP[Predicate +vc_to_inf]
  Semantics:
    * use(during(E), Agent, Theme, Predicate)
