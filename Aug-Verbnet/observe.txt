NP V NP to be NP (Infinitival Copular Clause)
  Syntax: NP[Agent] VERB NP[Theme] NP[Predicate]
  Semantics:
    * declare(during(E), Agent, Theme, Predicate)
NP V that S (S)
  Syntax: NP[Agent] VERB NP[Theme +that_comp]
  Semantics:
    * declare(during(E), Agent, Theme, ?Predicate)
NP V NP (Basic Transitive)
  Syntax: NP[Agent] VERB NP[Theme -sentential]
  Semantics:
    * declare(during(E), Agent, Theme, ?Predicate)
NP V NP.location PP.theme (NP-PP; Location Object, Theme-PP)
  Syntax: NP[Agent] VERB NP[Location] PREP[for] NP[Theme]
  Semantics:
    * search(during(E), Agent, Location, Theme)
NP V NP.location (Transitive; Location Object)
  Syntax: NP[Agent] VERB NP[Location]
  Semantics:
    * search(during(E), Agent, Location, ?Theme)
NP V that S (S)
  Syntax: NP[Agent] VERB NP[Topic +that_comp]

    * transfer_info(during(E), Agent, ?Recipient, Topic)
    * cause(Agent, E)

NP V NP (Basic Transitive)
  Syntax: NP[Experiencer] VERB NP[Stimulus]
  Semantics:
    * perceive(during(E), Experiencer, Stimulus)
