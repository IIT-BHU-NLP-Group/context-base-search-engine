NP V (Basic Intransitive)
  Syntax: NP[Theme] VERB
  Semantics:
    * appear(during(E), Theme)
NP V PP.location (PP; Location-PP)
  Syntax: NP[Theme] VERB PREP[+loc] NP[Location]
  Semantics:
    * appear(during(E), Theme)
    * Prep(start(E), Theme, Location)
    * Prep(end(E), Theme, Location)
There V NP PP (NP-PP; Expletive-there Subject)
  Syntax: LEX[there] VERB NP[Theme] PREP[+loc] NP[Location]
  Semantics:
    * appear(during(E), Theme)
    * Prep(start(E), Theme, Location)
    * Prep(end(E), Theme, Location)
PP.location V NP (Locative Inversion)
  Syntax: PREP[+loc] NP[Location] VERB NP[Theme]
  Semantics:
    * appear(during(E), Theme)
    * Prep(start(E), Theme, Location)
    * Prep(end(E), Theme, Location)
NP V ADV (ADVP-PRED; here/there)
  Syntax: NP[Theme] VERB NP[Location +adv_loc]
  Semantics:
    * appear(during(E), Theme)
    * location(start(E), Theme, Location)
    * location(end(E), Theme, Location)
NP V NP (Basic Transitive)
  Syntax: NP[Agent] VERB NP[Result]
  Semantics:
    * exist(start(E), Result)
    * exist(result(E), Result)
    * cause(Agent, E)
NP V NP PP.material (NP-PP; Material-PP)
  Syntax: NP[Agent] VERB NP[Result] PREP[from out_of] NP[Material]
  Semantics:
    * exist(start(E), Result)
    * exist(result(E), Result)
    * made_of(result(E), Result, Material)
    * cause(Agent, E)
NP V NP PP.beneficiary (NP-PP; for-PP)
  Syntax: NP[Agent] VERB NP[Result] PREP[for] NP[Beneficiary]
  Semantics:
    * exist(start(E), Result)
    * exist(result(E), Result)
    * cause(Agent, E)
    * benefit(E, Beneficiary)
NP V NP PP.attribute (NP-PP; as-PP)
  Syntax: NP[Agent] VERB NP[Result] LEX[as] NP[Attribute]
  Semantics:
    * exist(start(E), Result)
    * exist(result(E), Result)
    * cause(E, Agent)
NP V NP (NP)
  Syntax: NP[Agent] VERB NP[Theme]
  Semantics:
    * conclude(during(E), Agent, Theme, ?Source)
NP V NP PP.source (NP-PP; from-PP)
  Syntax: NP[Agent] VERB NP[Theme] PREP[from] NP[Source]
  Semantics:
    * conclude(during(E), Agent, Theme, Source)
NP V that S PP.theme (PP-THAT-S; from-PP)
  Syntax: NP[Agent] VERB PREP[from] NP[Theme +that_comp] NP[Source]
  Semantics:
    * conclude(during(E), Agent, Theme, Source)
NP V that S (S)
  Syntax: NP[Agent] VERB NP[Theme +that_comp]
  Semantics:
    * conclude(during(E), Agent, Theme, ?Source)
