NP V NP (NP)
  Syntax: NP[Agent] VERB NP[Theme]
  Semantics:
    * cause(Agent, E)
    * location(start(E), Theme)
    * location(end(E), Theme)
NP V NP PP.attribute (NP-PP; as-PP)
  Syntax: NP[Agent] VERB NP[Theme] PREP[as] NP[Attribute -sentential]
  Semantics:
    * cause(Agent, E)
    * location(start(E), Theme, Attribute)
    * location(end(E), Theme, Attribute)
NP V NP.patient (Basic Transitive; Causative)
  Syntax: NP[Agent] VERB NP[Patient]
  Semantics:
    * cause(Agent, E)
    * state(result(E), Endstate, Patient)
NP V NP PP.instrument (NP-PP; Instrument-PP)
  Syntax: NP[Agent] VERB NP[Patient] PREP[with] NP[Instrument]
  Semantics:
    * cause(Agent, E)
    * state(result(E), Endstate, Patient)
    * use(during(E), Agent, Instrument)
NP.patient V (Intransitive; Inchoative)
  Syntax: NP[Patient] VERB
  Semantics:
    * state(result(E), Endstate, Patient)
NP V ADV-Middle (Middle Construction)
  Syntax: NP[Patient] VERB ADV
  Semantics:
    * state(result(E), Endstate, Patient)
NP.instrument V NP (Instrument Subject Alternation)
  Syntax: NP[Instrument] VERB NP[Patient]
  Semantics:
    * use(during(E), ?Agent, Instrument)
    * state(result(E), Endstate, Patient)
NP V (Basic Intransitive)
  Syntax: NP[Agent +plural] VERB
  Semantics:
    * agree(result(E), Agent, ?Goal)
NP V PP.goal what S_INF (P-WHAT-TO-INF; on-PP)
  Syntax: NP[Agent +plural] VERB PREP[on] NP[Goal +what_inf]
  Semantics:
    * agree(result(E), Agent, Goal)
NP V PP.co-agent PP.goal (PP-PP; with-PP on-PP)
  Syntax: NP[Agent] VERB PREP[with] NP[Co-Agent] PREP[on] NP[Goal -sentential]
  Semantics:
    * agree(result(E), Agent, Co-Agent, Goal)
